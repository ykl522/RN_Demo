# 使用组件版本
    "react-native": "0.63.2"
    "react-navigation": "^4.0.10"

# 导入后在根目录执行 yarn 命令
    yarn/yarn install
    iOS 修改 bundle id 包名 下载profile文件后 cd ios 执行 pod install
    yarn start
    react-native run-android / react-native run-ios
